TABELLA AUTO IN VENDITA NEL CONCESSIONARIO

****************************************************

ID - MEDIUMINT PRIMARY_KEY AUTO_INCREMENT
TELAIO - VARCHAR (20) PRIMARY_KEY
FOTO - TEXT - DEFAULT ("../img/auto-DEFAULT.jpg) 
BRAND - VARCHAR (20) NOT_NULL
MODELLO - VARCHAR (40) NOT_NULL
KM.PERCORSI - MEDIUMINT NOT_NULL
PORTE - TINYINT
POSTI A SEDERE - TINYINT 
PREZZO DI VENDITA - FLOAT (8,2) NOT_NULL
PREZZO DI SCONTO - FLOAT (8,2)  
ANNO DI IMMATRICOLAZIONE - YEAR NOT_NULL
CARBURANTE - VARCHAR (40) NOT_NULL
OPTIONAL - TEXT
DESCRIZIONE - TEXT
CAMBIO - VARCHAR( 20 ) NOT_NULL
CLASSE DI EMISSIONI - VARCHAR(20) NOT_NULL
TARGA - VARCHAR (20) PRIMARY_KEY
CILINDRATA - FLOAT(4.1) NOT_NULL
PROPRIETARI - TINYINT 
COLORE - VARCHAR (30)
